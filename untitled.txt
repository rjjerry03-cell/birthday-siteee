<script>  
    // 1. GSAP Animations on Load  
    window.addEventListener('load', () => {  
        gsap.to("#main-card", {  
            duration: 1.5,  
            y: 0,  
            opacity: 1,  
            ease: "power3.out"  
        });  
    });  

    // 2. Music Player Logic  
    const musicBtn = document.getElementById('music-btn');  
    const audio = document.getElementById('bg-music');  
    let isPlaying = false;  

    musicBtn.addEventListener('click', () => {  
        if (isPlaying) {  
            audio.pause();  
            musicBtn.innerHTML = '<i class="fas fa-music"></i>';  
            musicBtn.classList.remove('text-pink-400');  
        } else {  
            audio.play();  
            musicBtn.innerHTML = '<i class="fas fa-pause"></i>';  
            musicBtn.classList.add('text-pink-400');  
        }  
        isPlaying = !isPlaying;  
    });  

    // 3. Floating Hearts Generator  
    function createHeart() {  
        const heart = document.createElement('div');  
        heart.classList.add('heart');  
        heart.innerHTML = '<i class="fas fa-heart"></i>';  
          
        // Randomize position and size  
        heart.style.left = Math.random() * 100 + 'vw';  
        heart.style.fontSize = Math.random() * 20 + 10 + 'px';  
          
        // Randomize animation duration  
        heart.style.animationDuration = Math.random() * 3 + 4 + 's'; // 4 to 7 seconds  
          
        document.getElementById('hearts-container').appendChild(heart);  

        // Remove heart after animation  
        setTimeout(() => {  
            heart.remove();  
        }, 7000);  
    }  

    // Generate hearts every 300ms  
    setInterval(createHeart, 300);  

    // 4. Make a Wish Interaction  
    function makeWish() {  
        const msg = document.getElementById('wish-msg');  
        msg.style.height = 'auto';  
        msg.style.opacity = '1';  
          
        // Burst of hearts on click  
        for(let i=0; i<30; i++) {  
            setTimeout(createHeart, i * 50);  
        }  
          
        confettiEffect();  
    }  

    // Simple confetti using existing heart logic but faster  
    function confettiEffect() {  
        for(let i=0; i<15; i++) {  
            const heart = document.createElement('div');  
            heart.classList.add('heart');  
            heart.innerHTML = 'âœ¨';  
            heart.style.left = '50%';  
            heart.style.bottom = '50%';  
            heart.style.fontSize = '20px';  
            heart.style.animation = 'none'; // reset  
              
            // Explode animation manually  
            const angle = Math.random() * Math.PI * 2;  
            const velocity = Math.random() * 200 + 50;  
              
            gsap.to(heart, {  
                x: Math.cos(angle) * velocity,  
                y: Math.sin(angle) * velocity,  
                opacity: 0,  
                duration: 1.5,  
                onComplete: () => heart.remove()  
            });  
              
            document.getElementById('hearts-container').appendChild(heart);  
        }  
    }  
</script>
